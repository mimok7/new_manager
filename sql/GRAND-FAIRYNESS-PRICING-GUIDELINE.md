### SELECT로 검증 (쿼리 실행)

```sql
-- 총 행수 확인
SELECT COUNT(*) FROM cruise_rate_card 
WHERE cruise_name = '그랜드 파이어니스 크루즈' AND valid_year = 2026;
-- 결과: 20행 ✓

-- 시즌별 객실 수 확인
SELECT COUNT(*), schedule_type FROM cruise_rate_card 
WHERE cruise_name = '그랜드 파이어니스 크루즈' AND valid_year = 2026
GROUP BY schedule_type;
-- 결과: S1=10, S2=10 ✓

-- 시즌 1 일반실 아동가 확인 (3,200,000 고정)
SELECT room_type, price_adult, price_child
FROM cruise_rate_card 
WHERE cruise_name = '그랜드 파이어니스 크루즈' 
  AND valid_year = 2026
  AND schedule_type = '1N2D-S1'
  AND room_type IN ('오션스위트룸', '오션발코니 스위트');
-- price_child = 3,200,000 확인 ✓

-- 시즌 2 가격 인하 확인
SELECT room_type, 
       MAX(CASE WHEN schedule_type = '1N2D-S1' THEN price_adult END) as S1성인,
       MAX(CASE WHEN schedule_type = '1N2D-S2' THEN price_adult END) as S2성인
FROM cruise_rate_card 
WHERE cruise_name = '그랜드 파이어니스 크루즈' AND valid_year = 2026
GROUP BY room_type
ORDER BY room_type;
-- 인하된 객실 S1 > S2 확인 ✓

-- 프리미엄 객실 아동요금 = 성인요금 확인
SELECT room_type, price_adult, price_child
FROM cruise_rate_card 
WHERE cruise_name = '그랜드 파이어니스 크루즈' 
  AND valid_year = 2026
  AND room_type LIKE '%오너스%';
-- price_adult = price_child 확인 ✓
```
